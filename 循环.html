/* 多个for循环的问题考究 */

#include

int main()

{

int i;

int j;

for (i=1; i<10; i++) //1.i=1 1-9<10 3.

{

{

for (j=1; j < i+1; j++) //2.j=1,1<2到10,

printf("%d*%d=%d\t",j,i,i*j);//3.1*1=1 1*2=2

}

printf("\n");

}

return 0;

}
